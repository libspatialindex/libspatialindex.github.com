00174 double MovingPoint::getCoord(uint32_t d, double t) const
00175 {
00176     if (d >= m_dimension) throw Tools::IndexOutOfBoundsException(d);
00177 
00178     if (t >= m_endTime) return m_pCoords[d] + m_pVCoords[d] * (m_endTime - m_startTime);
00179     else if (t <= m_startTime) return m_pCoords[d] + m_pVCoords[d] * m_startTime;
00180     else return m_pCoords[d] + m_pVCoords[d] * (t - m_startTime);
