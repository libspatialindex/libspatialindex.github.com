128 uint32_t Statistics::getTreeHeight() const
129 {
130  uint32_t ret = 0;
131 
132  for (size_t cIndex = 0; cIndex < m_treeHeight.size(); ++cIndex)
133  {
134  ret = std::max(ret, m_treeHeight[cIndex]);
135  }
136 
137  return ret;
