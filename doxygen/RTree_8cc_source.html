00740 bool SpatialIndex::RTree::RTree::isIndexValid()
00741 {
00742     bool ret = true;
00743     std::stack<ValidateEntry> st;
00744     NodePtr root = readNode(m_rootID);
00745 
00746     if (root->m_level != m_stats.m_u32TreeHeight - 1)
00747     {
00748         std::cerr << "Invalid tree height." << std::endl;
00749         return false;
00750     }
00751 
00752     std::map<uint32_t, uint32_t> nodesInLevel;
00753     nodesInLevel.insert(std::pair<uint32_t, uint32_t>(root->m_level, 1));
00754 
00755     ValidateEntry e(root->m_nodeMBR, root);
00756     st.push(e);
00757 
00758     while (! st.empty())
00759     {
00760         e = st.top(); st.pop();
00761 
00762         Region tmpRegion;
00763         tmpRegion = m_infiniteRegion;
00764 
00765         for (uint32_t cDim = 0; cDim < tmpRegion.m_dimension; ++cDim)
00766         {
00767             tmpRegion.m_pLow[cDim] = std::numeric_limits<double>::max();
00768             tmpRegion.m_pHigh[cDim] = -std::numeric_limits<double>::max();
00769 
00770             for (uint32_t cChild = 0; cChild < e.m_pNode->m_children; ++cChild)
00771             {
00772                 tmpRegion.m_pLow[cDim] = std::min(tmpRegion.m_pLow[cDim], e.m_pNode->m_ptrMBR[cChild]->m_pLow[cDim]);
00773                 tmpRegion.m_pHigh[cDim] = std::max(tmpRegion.m_pHigh[cDim], e.m_pNode->m_ptrMBR[cChild]->m_pHigh[cDim]);
00774             }
00775         }
00776 
00777         if (! (tmpRegion == e.m_pNode->m_nodeMBR))
00778         {
00779             std::cerr << "Invalid parent information." << std::endl;
00780             ret = false;
00781         }
00782         else if (! (tmpRegion == e.m_parentMBR))
00783         {
00784             std::cerr << "Error in parent." << std::endl;
00785             ret = false;
00786         }
00787 
00788         if (e.m_pNode->m_level != 0)
00789         {
00790             for (uint32_t cChild = 0; cChild < e.m_pNode->m_children; ++cChild)
00791             {
00792                 NodePtr ptrN = readNode(e.m_pNode->m_pIdentifier[cChild]);
00793                 ValidateEntry tmpEntry(*(e.m_pNode->m_ptrMBR[cChild]), ptrN);
00794 
00795                 std::map<uint32_t, uint32_t>::iterator itNodes = nodesInLevel.find(tmpEntry.m_pNode->m_level);
00796 
00797                 if (itNodes == nodesInLevel.end())
00798                 {
00799                     nodesInLevel.insert(std::pair<uint32_t, uint32_t>(tmpEntry.m_pNode->m_level, 1l));
00800                 }
00801                 else
00802                 {
00803                     nodesInLevel[tmpEntry.m_pNode->m_level] = nodesInLevel[tmpEntry.m_pNode->m_level] + 1;
00804                 }
00805 
00806                 st.push(tmpEntry);
00807             }
00808         }
00809     }
00810 
00811     uint32_t nodes = 0;
00812     for (uint32_t cLevel = 0; cLevel < m_stats.m_u32TreeHeight; ++cLevel)
00813     {

00815         {
00816             std::cerr << "Invalid nodesInLevel information." << std::endl;
00817             ret = false;
00818         }
00819 
00820         nodes += m_stats.m_nodesInLevel[cLevel];
00821     }
00822 
00823     if (nodes != m_stats.m_u32Nodes)
00824     {
00825         std::cerr << "Invalid number of nodes information." << std::endl;
00826         ret = false;
00827     }
00828 
00829     return ret;
00830 }
