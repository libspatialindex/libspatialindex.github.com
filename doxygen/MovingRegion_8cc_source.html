00328 // if tmin, tmax are infinity then this will not work correctly (everything will always intersect).
00329 // does not work for shrinking regions.
00330 // does not work with degenerate time-intervals.
00331 //
00332 // WARNING: this will return true even if one region completely contains the other, since
00333 // their areas do intersect in that case!
00334 //
00335 // there are various cases here:
00336 // 1. one region contains the other.
00337 // 2. one boundary of one region is always contained indide the other region, while the other
00338 //    boundary is not (so no boundaries will ever intersect).
00339 // 3. either the upper or lower boundary of one region intersects a boundary of the other.
00340 bool MovingRegion::intersectsRegionInTime(const IInterval& ivPeriod, const MovingRegion& r, IInterval& ivOut) const
00341 {
00342     if (m_dimension != r.m_dimension) throw Tools::IllegalArgumentException("intersectsRegionInTime: MovingRegions have different number of dimensions.");
00343 
00344     assert(m_startTime < m_endTime);
00345     assert(r.m_startTime < r.m_endTime);
00346     assert(ivPeriod.getLowerBound() < ivPeriod.getUpperBound());
00347     assert(isShrinking() == false && r.isShrinking() == false);
00348 
00349     // this is needed, since we are assuming below that the two regions have some point of intersection
00350     // inside itPeriod.
00351     if (containsRegionInTime(ivPeriod, r) || r.containsRegionInTime(ivPeriod, *this))
00352     {
00353         ivOut = ivPeriod;
00354         return true;
00355     }
00356 
00357     double tmin = std::max(m_startTime, r.m_startTime);
00358     double tmax = std::min(m_endTime, r.m_endTime);
00359 
00360     // the regions do not intersect in time.
00361     if (tmax <= tmin) return false;
00362 
00363     tmin = std::max(tmin, ivPeriod.getLowerBound());
00364     tmax = std::min(tmax, ivPeriod.getUpperBound());
00365 
00366     // the regions intersecting interval does not intersect with the given time period.
00367     if (tmax <= tmin) return false;
00368 
00369     assert(tmax < std::numeric_limits<double>::max());
00370     assert(tmin > -std::numeric_limits<double>::max());
00371 
00372     // I use projected low and high because they are faster and it does not matter.
00373     // The are also necessary for calculating the intersection point with reference time instant 0.0.
00374 
00375     for (uint32_t cDim = 0; cDim < m_dimension; ++cDim)
00376     {
00377     assert(
00378         tmin >= ivPeriod.getLowerBound() && tmax <= ivPeriod.getUpperBound() &&
00379         tmin >= m_startTime && tmax <= m_endTime &&
00380         tmin >= r.m_startTime && tmax <= r.m_endTime);
00381 
00382         // completely above or bellow in i-th dimension
00383         if (
00384             (r.getExtrapolatedLow(cDim, tmin) > getExtrapolatedHigh(cDim, tmin) &&
00385             r.getExtrapolatedLow(cDim, tmax) >= getExtrapolatedHigh(cDim, tmax)) ||
00386             (r.getExtrapolatedHigh(cDim, tmin) < getExtrapolatedLow(cDim, tmin) &&
00387             r.getExtrapolatedHigh(cDim, tmax) <= getExtrapolatedLow(cDim, tmax)))
00388             return false;
00389 
00390         // otherwise they intersect inside this interval for sure. Care needs to be taken since
00391         // intersection does not necessarily mean that two line segments intersect. It could be
00392         // that one line segment is completely above/below another, in which case there is no intersection
00393         // point inside tmin, tmax, even though the two region areas do intersect.
00394 
00395         if (r.getExtrapolatedLow(cDim, tmin) > getExtrapolatedHigh(cDim, tmin))  // r above *this at tmin
00396         {
00397             tmin = (getExtrapolatedHigh(cDim, 0.0) - r.getExtrapolatedLow(cDim, 0.0)) / (r.getVLow(cDim) - getVHigh(cDim));
00398         }
00399         else if (r.getExtrapolatedHigh(cDim, tmin) < getExtrapolatedLow(cDim, tmin)) // r below *this at tmin
00400         {
00401             tmin = (getExtrapolatedLow(cDim, 0.0) - r.getExtrapolatedHigh(cDim, 0.0)) / (r.getVHigh(cDim) - getVLow(cDim));
00402         }
00403         // else they do not intersect and the boundary might be completely contained in this region.
00404 
00405         if (r.getExtrapolatedLow(cDim, tmax) > getExtrapolatedHigh(cDim, tmax))  // r above *this at tmax
00406         {
00407             tmax = (getExtrapolatedHigh(cDim, 0.0) - r.getExtrapolatedLow(cDim, 0.0)) / (r.getVLow(cDim) - getVHigh(cDim));
00408         }
00409         else if (r.getExtrapolatedHigh(cDim, tmax) < getExtrapolatedLow(cDim, tmax)) // r below *this at tmax
00410         {
00411             tmax = (getExtrapolatedLow(cDim, 0.0) - r.getExtrapolatedHigh(cDim, 0.0)) / (r.getVHigh(cDim) - getVLow(cDim));
00412         }
00413         // else they do not intersect and the boundary might be completely contained in this region.
00414 
00415         assert(tmin <= tmax);
00416     }
00417 
00418     assert(
00419         tmin >= ivPeriod.getLowerBound() && tmax <= ivPeriod.getUpperBound() &&
00420         tmin >= m_startTime && tmax <= m_endTime &&
00421         tmin >= r.m_startTime && tmax <= r.m_endTime);
00422 
00423     ivOut.setBounds(tmin, tmax);
00424 
00425     return true;

00434 // works fine for infinite bounds (both tmin and tmax).

00436 //
00437 // finds if during the intersecting time-interval of r and ivPeriod, r is completely contained in *this.

00441

00445     double tmax = std::min(ivPeriod.getUpperBound(), r.m_endTime);

00457         tmin >= m_startTime && tmax <= m_endTime &&

00464             r.getExtrapolatedLow(cDim, tmin) < getExtrapolatedLow(cDim, tmin)) return false;
00465

00467         if (r.m_pVHigh[cDim] != m_pVHigh[cDim])

00473         }

00476         {

00502 double MovingRegion::getProjectedSurfaceAreaInTime(const IInterval& ivI) const
00503 {

00505     double tmax = std::min(ivI.getUpperBound(), m_endTime);
00506 
00507     assert(tmin > -std::numeric_limits<double>::max());
00508     assert(tmax < std::numeric_limits<double>::max());
00509     assert(tmin <= tmax);
00510 
00511     if (tmin >= tmax - std::numeric_limits<double>::epsilon() &&
00512         tmin <= tmax + std::numeric_limits<double>::epsilon())
00513         return 0.0;
00514 
00515     double dx1, dx2, dx3;
00516     double dv1, dv2, dv3;
00517     double H = tmax - tmin;
00518 
00519     if (m_dimension == 3)
00520     {
00521         dx3 = getExtrapolatedHigh(2, tmin) - getExtrapolatedLow(2, tmin);
00522         dv3 = getVHigh(2) - getVLow(2);
00523         dx2 = getExtrapolatedHigh(1, tmin) - getExtrapolatedLow(1, tmin);
00524         dv2 = getVHigh(1) - getVLow(1);
00525         dx1 = getExtrapolatedHigh(0, tmin) - getExtrapolatedLow(0, tmin);
00526         dv1 = getVHigh(0) - getVLow(0);
00527         return
00528             H * (dx1 + dx2 + dx3 + dx1*dx2 + dx1*dx3 + dx2*dx3) +
00529             H*H * (dv1 + dv2 + dv3 + dx1*dv2 + dv1*dx2 + dx1*dv3 +
00530             dv1*dx3 + dx2*dv3 + dv2*dx3) / 2.0 +
00531             H*H*H * (dv1*dv2 + dv1*dv3 + dv2*dv3) / 3.0;
00532     }
00533     else if (m_dimension == 2)

00538         dv1 = getVHigh(0) - getVLow(0);

00540     }

00543         // marioh: why not use the increase of the length of the interval here?
00544         return 0.0;

00546     else

00549     }

00558 double MovingRegion::getCenterDistanceInTime(const IInterval& ivI, const MovingRegion& r) const
00559 {
00560     if (m_dimension != r.m_dimension) throw Tools::IllegalArgumentException("getCenterDistanceInTime: MovingRegions have different number of dimensions.");
00561 
00562     assert(m_startTime < m_endTime);
00563     assert(r.m_startTime < r.m_endTime);
00564     assert(ivI.getLowerBound() < ivI.getUpperBound());
00565 
00566     double tmin = std::max(m_startTime, r.m_startTime);
00567     double tmax = std::min(m_endTime, r.m_endTime);
00568 
00569     // the regions do not intersect in time.
00570     if (tmax <= tmin) return 0.0;
00571 
00572     tmin = std::max(tmin, ivI.getLowerBound());
00573     tmax = std::min(tmax, ivI.getUpperBound());
00574 
00575     // the regions intersecting interval does not intersect with the given time period.
00576     if (tmax <= tmin) return 0.0;
00577 
00578     assert(tmax < std::numeric_limits<double>::max());
00579     assert(tmin > -std::numeric_limits<double>::max());
00580 
00581     if (tmin >= tmax - std::numeric_limits<double>::epsilon() &&
00582         tmin <= tmax + std::numeric_limits<double>::epsilon())
00583         return 0.0;
00584 
00585     double H = tmax - tmin;
00586 
00587     double* dx = new double[m_dimension];
00588     double* dv = new double[m_dimension];
00589     double a = 0.0, b = 0.0, c = 0.0, f = 0.0, l = 0.0, m = 0.0, n = 0.0;
00590 
00591     for (uint32_t cDim = 0; cDim < m_dimension; ++cDim)
00592     {
00593         dx[cDim] =
00594             (r.getExtrapolatedLow(cDim, tmin) + r.getExtrapolatedHigh(cDim, tmin)) / 2.0 -
00595             (getExtrapolatedLow(cDim, tmin) + getExtrapolatedHigh(cDim, tmin)) / 2.0;
00596         dv[cDim] =
00597             (r.getVLow(cDim) + r.getVHigh(cDim)) / 2.0 -
00598             (getVLow(cDim) + getVHigh(cDim)) / 2.0;
00599     }
00600 
00601     for (uint32_t cDim = 0; cDim < m_dimension; ++cDim)
00602     {
00603         a += dv[cDim] * dv[cDim];
00604         b += 2.0 * dx[cDim] * dv[cDim];
00605         c += dx[cDim] * dx[cDim];
00606     }
00607 
00608     if (a == 0.0 && c == 0.0) return 0.0;
00609     if (a == 0.0) return H * std::sqrt(c);
00610     if (c == 0.0) return H * H * std::sqrt(a) / 2.0;
00611 
00612     f = std::sqrt(a * H * H + b * H + c);
00613     l = 2.0 * a * H + b;
00614     m = 4.0 * a * c - b * b;
00615     n = 2.0 * std::sqrt(a);
00616 
00617     delete[] dx;
00618     delete[] dv;
00619 
00620     return (l * f + log(l / n + f) * m / n - b * std::sqrt(c) - std::log(b / n + std::sqrt(c)) * m / n) / (4.0 * a);
00621 }

01101 // this computes the area/volume/etc. swept by the Region in time.
01102 double MovingRegion::getAreaInTime(const IInterval& ivI) const
01103 {
01104     double tmin = std::max(ivI.getLowerBound(), m_startTime);
01105     double tmax = std::min(ivI.getUpperBound(), m_endTime);
01106 
01107     assert(tmin > -std::numeric_limits<double>::max());
01108     assert(tmax < std::numeric_limits<double>::max());
01109     assert(tmin <= tmax);
01110 
01111     if (tmin >= tmax - std::numeric_limits<double>::epsilon() &&
01112         tmin <= tmax + std::numeric_limits<double>::epsilon())
01113         return 0.0;
01114 
01115     double dx1, dx2, dx3;
01116     double dv1, dv2, dv3;
01117     double H = tmax - tmin;
01118 
01119     if (m_dimension == 3)
01120     {
01121         dx3 = getExtrapolatedHigh(2, tmin) - getExtrapolatedLow(2, tmin);
01122         dv3 = getVHigh(2) - getVLow(2);
01123         dx2 = getExtrapolatedHigh(1, tmin) - getExtrapolatedLow(1, tmin);
01124         dv2 = getVHigh(1) - getVLow(1);
01125         dx1 = getExtrapolatedHigh(0, tmin) - getExtrapolatedLow(0, tmin);
01126         dv1 = getVHigh(0) - getVLow(0);
01127         return
01128             H * dx1 * dx2 * dx3 + H * H * (dx1 * dx2 * dv3 + (dx1 * dv2 + dv1 * dx2) * dx3) / 2.0 +
01129             H * H * H * ((dx1 * dv2 + dv1 * dx2) * dv3 + dv1 * dv2 * dx3) / 3.0 + H * H * H * H * dv1 * dv2 * dv3 / 4.0;
01130     }
01131     else if (m_dimension == 2)
01132     {
01133         dx2 = getExtrapolatedHigh(1, tmin) - getExtrapolatedLow(1, tmin);
01134         dv2 = getVHigh(1) - getVLow(1);
01135         dx1 = getExtrapolatedHigh(0, tmin) - getExtrapolatedLow(0, tmin);
01136         dv1 = getVHigh(0) - getVLow(0);
01137         return H * dx1 * dx2 + H * H * (dx1 * dv2 + dv1 * dx2) / 2.0 + H * H * H * dv1 * dv2 / 3.0;
01138     }
01139     else if (m_dimension == 1)
01140     {
01141         dx1 = getExtrapolatedHigh(0, tmin) - getExtrapolatedLow(0, tmin);
01142         dv1 = getVHigh(0) - getVLow(0);
01143         return H * dx1 + H * H * dv1 / 2.0;
01144     }
01145     else
01146     {
01147         throw Tools::NotSupportedException("getAreaInTime: unsupported dimensionality.");
01148     }
01149 }
