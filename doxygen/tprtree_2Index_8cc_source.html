00138     Tools::Interval ivT(m_pTree->m_currentTime, m_pTree->m_currentTime + m_pTree->m_horizon);
00139

00141     {
00142         // I need the combined region from current time up to infinity here.
00143         m_ptrMBR[cChild]->getCombinedRegionAfterTime(ivT.getLowerBound(), *t, r);
00144 
00145         double a = m_ptrMBR[cChild]->getAreaInTime(ivT);
00146         double b = t->getAreaInTime(ivT);
00147         double enl = b - a;
00148 
00149         if (enl < area)
00150         {
00151             area = enl;
00152             best = cChild;
00153         }
00154         else if (enl == area)
00155         {
00156             // this will rarely happen, so compute best area on the fly only
00157             // when necessary.
00158             if (a < m_ptrMBR[best]->getAreaInTime(ivT)) best = cChild;
00159         }
00160     }
