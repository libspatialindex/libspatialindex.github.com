01284 
01285     const Tools::IInterval* ti = dynamic_cast<const Tools::IInterval*>(&query);
01286     if (ti == 0) throw Tools::IllegalArgumentException("rangeQuery: Shape does not support the Tools::IInterval interface.");

01295     try
01296     {

01307         }
