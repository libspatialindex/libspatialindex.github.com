88  void release(TPRTree::Node* p)
89  {
90  if (p != nullptr)
91  {
92  if (m_pool.size() < m_capacity)
93  {
94  if (p->m_pData != nullptr)
95  {
96  for (uint32_t cChild = 0; cChild < p->m_children; ++cChild)
97  {
98  if (p->m_pData[cChild] != nullptr) delete[] p->m_pData[cChild];
99  }
100  }
101 
102  p->m_level = 0;
103  p->m_identifier = -1;
104  p->m_children = 0;
105  p->m_totalDataLength = 0;
106 
107  m_pool.push(p);
108  }
109  else
110  {
111  #ifndef NDEBUG
112  --m_pointerCount;
113  #endif
114  delete p;
115  }
116 
117  assert(m_pool.size() <= m_capacity);
118  }
119  }
