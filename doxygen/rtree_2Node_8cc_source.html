00928 void Node::pickSeeds(uint32_t& index1, uint32_t& index2)
00929 {
00930     double separation = -std::numeric_limits<double>::max();
00931     double inefficiency = -std::numeric_limits<double>::max();
00932     uint32_t cDim, u32Child, cIndex;
00933 
00934     switch (m_pTree->m_treeVariant)
00935     {
00936         case RV_LINEAR:
00937         case RV_RSTAR:
00938             for (cDim = 0; cDim < m_pTree->m_dimension; ++cDim)
00939             {
00940                 double leastLower = m_ptrMBR[0]->m_pLow[cDim];
00941                 double greatestUpper = m_ptrMBR[0]->m_pHigh[cDim];
00942                 uint32_t greatestLower = 0;
00943                 uint32_t leastUpper = 0;
00944                 double width;
00945 
00946                 for (u32Child = 1; u32Child <= m_capacity; ++u32Child)
00947                 {
00948                     if (m_ptrMBR[u32Child]->m_pLow[cDim] > m_ptrMBR[greatestLower]->m_pLow[cDim]) greatestLower = u32Child;
00949                     if (m_ptrMBR[u32Child]->m_pHigh[cDim] < m_ptrMBR[leastUpper]->m_pHigh[cDim]) leastUpper = u32Child;
00950 
00951                     leastLower = std::min(m_ptrMBR[u32Child]->m_pLow[cDim], leastLower);
00952                     greatestUpper = std::max(m_ptrMBR[u32Child]->m_pHigh[cDim], greatestUpper);
00953                 }
00954 
00955                 width = greatestUpper - leastLower;
00956                 if (width <= 0) width = 1;
00957 
00958                 double f = (m_ptrMBR[greatestLower]->m_pLow[cDim] - m_ptrMBR[leastUpper]->m_pHigh[cDim]) / width;
00959 
00960                 if (f > separation)
00961                 {
00962                     index1 = leastUpper;
00963                     index2 = greatestLower;
00964                     separation = f;
00965                 }
00966             }  // for (cDim)
00967 
00968             if (index1 == index2)
00969             {
00970                 if (index2 == 0) ++index2;
00971                 else --index2;
00972             }
00973 
00974             break;
00975         case RV_QUADRATIC:
00976             // for each pair of Regions (account for overflow Region too!)
00977             for (u32Child = 0; u32Child < m_capacity; ++u32Child)
00978             {
00979                 double a = m_ptrMBR[u32Child]->getArea();
00980 
00981                 for (cIndex = u32Child + 1; cIndex <= m_capacity; ++cIndex)
00982                 {
00983                     // get the combined MBR of those two entries.
00984                     Region r;
00985                     m_ptrMBR[u32Child]->getCombinedRegion(r, *(m_ptrMBR[cIndex]));
00986 
00987                     // find the inefficiency of grouping these entries together.
00988                     double d = r.getArea() - a - m_ptrMBR[cIndex]->getArea();
00989 
00990                     if (d > inefficiency)
00991                     {
00992                         inefficiency = d;
00993                         index1 = u32Child;
00994                         index2 = cIndex;
00995                     }
00996                 }  // for (cIndex)
00997             } // for (u32Child)
00998 
00999             break;
01000         default:
01001             throw Tools::NotSupportedException("Node::pickSeeds: Tree variant not supported.");
01002     }
