00516     if ((m_low > low && m_low < high) || (m_high > low && m_high < high)) return true;

00518     switch (m_type)
00519     {
00520     case IT_CLOSED:
00521         if (m_low == high)
00522         {
00523             if (type == IT_CLOSED || type == IT_LEFTOPEN) return true;
00524             else return false;

00528             if (type == IT_CLOSED || type == IT_RIGHTOPEN) return true;
