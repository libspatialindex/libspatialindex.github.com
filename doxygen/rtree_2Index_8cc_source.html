00140 {
00141     double area = std::numeric_limits<double>::max();
00142     uint32_t best = std::numeric_limits<uint32_t>::max();
00143 
00144     RegionPtr t = m_pTree->m_regionPool.acquire();
00145 
00146     for (uint32_t cChild = 0; cChild < m_children; ++cChild)
00147     {
00148         m_ptrMBR[cChild]->getCombinedRegion(*t, r);
00149 
00150         double a = m_ptrMBR[cChild]->getArea();
00151         double enl = t->getArea() - a;
00152 
00153         if (enl < area)
00154         {
00155             area = enl;
00156             best = cChild;
00157         }
00158         else if (enl == area)
00159         {
00160             // this will rarely happen, so compute best area on the fly only
00161             // when necessary.
00162             if (a < m_ptrMBR[best]->getArea()) best = cChild;
00163         }
00164     }

00169 uint32_t Index::findLeastOverlap(const Region& r) const
00170 {
00171     OverlapEntry** entries = new OverlapEntry*[m_children];
00172 
00173     double leastOverlap = std::numeric_limits<double>::max();
00174     double me = std::numeric_limits<double>::max();
00175     OverlapEntry* best = 0;
00176 
00177     // find combined region and enlargement of every entry and store it.
00178     for (uint32_t cChild = 0; cChild < m_children; ++cChild)
00179     {
00180         try
00181         {
00182             entries[cChild] = new OverlapEntry();
00183         }
00184         catch (...)
00185         {
00186             for (uint32_t i = 0; i < cChild; ++i) delete entries[i];
00187             delete[] entries;
00188             throw;
00189         }
00190 
00191         entries[cChild]->m_index = cChild;
00192         entries[cChild]->m_original = m_ptrMBR[cChild];
00193         entries[cChild]->m_combined = m_pTree->m_regionPool.acquire();
00194         m_ptrMBR[cChild]->getCombinedRegion(*(entries[cChild]->m_combined), r);
00195         entries[cChild]->m_oa = entries[cChild]->m_original->getArea();
00196         entries[cChild]->m_ca = entries[cChild]->m_combined->getArea();
00197         entries[cChild]->m_enlargement = entries[cChild]->m_ca - entries[cChild]->m_oa;
00198 
00199         if (entries[cChild]->m_enlargement < me)
00200         {
00201             me = entries[cChild]->m_enlargement;
00202             best = entries[cChild];
00203         }
00204         else if (entries[cChild]->m_enlargement == me && entries[cChild]->m_oa < best->m_oa)
00205         {
00206             best = entries[cChild];
00207         }
00208     }
00209 
00210     if (me < -std::numeric_limits<double>::epsilon() || me > std::numeric_limits<double>::epsilon())
00211     {
00212         uint32_t cIterations;
00213 
00214         if (m_children > m_pTree->m_nearMinimumOverlapFactor)
00215         {
00216             // sort entries in increasing order of enlargement.
00217             ::qsort(entries, m_children,
00218                             sizeof(OverlapEntry*),
00219                             OverlapEntry::compareEntries);
00220             assert(entries[0]->m_enlargement <= entries[m_children - 1]->m_enlargement);
00221 
00222             cIterations = m_pTree->m_nearMinimumOverlapFactor;
00223         }
00224         else
00225         {
00226             cIterations = m_children;
00227         }
00228 
00229         // calculate overlap of most important original entries (near minimum overlap cost).
00230         for (uint32_t cIndex = 0; cIndex < cIterations; ++cIndex)
00231         {
00232             double dif = 0.0;
00233             OverlapEntry* e = entries[cIndex];
00234 
00235             for (uint32_t cChild = 0; cChild < m_children; ++cChild)
00236             {
00237                 if (e->m_index != cChild)
00238                 {
00239                     double f = e->m_combined->getIntersectingArea(*(m_ptrMBR[cChild]));
00240                     if (f != 0.0) dif += f - e->m_original->getIntersectingArea(*(m_ptrMBR[cChild]));
00241                 }
00242             } // for (cChild)
00243 
00244             if (dif < leastOverlap)
00245             {
00246                 leastOverlap = dif;
00247                 best = entries[cIndex];
00248             }
00249             else if (dif == leastOverlap)
00250             {
00251                 if (e->m_enlargement == best->m_enlargement)
00252                 {
00253                     // keep the one with least area.
00254                     if (e->m_original->getArea() < best->m_original->getArea()) best = entries[cIndex];
00255                 }
00256                 else
00257                 {
00258                     // keep the one with least enlargement.
00259                     if (e->m_enlargement < best->m_enlargement) best = entries[cIndex];
00260                 }
00261             }
00262         } // for (cIndex)
00263     }
00264 
00265     uint32_t ret = best->m_index;
00266 
00267     for (uint32_t cChild = 0; cChild < m_children; ++cChild)
00268     {
00269         delete entries[cChild];
00270     }
00271     delete[] entries;
00272 
00273     return ret;
00274 }

00277 {
00278     ++(m_pTree->m_stats.m_u64Adjustments);

00280     // find entry pointing to old node;

00299         {

00312
