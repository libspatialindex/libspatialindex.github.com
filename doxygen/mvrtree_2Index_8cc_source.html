47 NodePtr Index::chooseSubtree(const TimeRegion& mbr, uint32_t insertionLevel, std::stack<id_type>& pathBuffer)
48 {
49  if (m_level == insertionLevel) return NodePtr(this, &(m_pTree->m_indexPool));
50 
51  pathBuffer.push(m_identifier);
52 
53  uint32_t child = 0;
54 
55  switch (m_pTree->m_treeVariant)
56  {
57  case RV_LINEAR:
58  case RV_QUADRATIC:
59  child = findLeastEnlargement(mbr);
60  break;
61  case RV_RSTAR:
62  if (m_level == 1)
63  {
64  // if this node points to leaves...
65  child = findLeastOverlap(mbr);
66  }
67  else
68  {
69  child = findLeastEnlargement(mbr);
70  }
71  break;
72  default:
73  throw Tools::NotSupportedException("Index::chooseSubtree: Tree variant not supported.");
74  }
75  assert (child != std::numeric_limits<uint32_t>::max());
76 
77  NodePtr n = m_pTree->readNode(m_pIdentifier[child]);
78  NodePtr ret = n->chooseSubtree(mbr, insertionLevel, pathBuffer);
79  assert(n.unique());
80  if (ret.get() == n.get()) n.relinquish();
81 
82  return ret;
83 }

97  {

153 uint32_t Index::findLeastEnlargement(const TimeRegion& r) const
154 {
155  double area = std::numeric_limits<double>::max();
156  uint32_t best = std::numeric_limits<uint32_t>::max();
157 
158  TimeRegionPtr t = m_pTree->m_regionPool.acquire();
159 
160  for (uint32_t cChild = 0; cChild < m_children; ++cChild)
161  {
162  // if this child is already dead do not consider it.
163  if (m_ptrMBR[cChild]->m_endTime <= r.m_startTime) continue;
164 
165  m_ptrMBR[cChild]->getCombinedRegion(*t, r);
166 
167  double a = m_ptrMBR[cChild]->getArea();
168  double enl = t->getArea() - a;
169 
170  if (enl < area)
171  {
172  area = enl;
173  best = cChild;
174  }
175  else if (
176  enl > area - std::numeric_limits<double>::epsilon() &&
177  enl < area + std::numeric_limits<double>::epsilon())
178  {
179  if (a < m_ptrMBR[best]->getArea()) best = cChild;
180  }
181  }
182 
183 #ifndef NDEBUG
184  if (best == std::numeric_limits<uint32_t>::max())
185  {
186  std::ostringstream s;
187  s << "findLeastEnlargement: All entries of node " << m_identifier << " are dead.";
188  throw Tools::IllegalStateException(s.str());
189  }
190 #endif
191 
192  return best;
193 }

195 uint32_t Index::findLeastOverlap(const TimeRegion& r) const
196 {
197  OverlapEntry** entries = new OverlapEntry*[m_children];
198 
199  double leastOverlap = std::numeric_limits<double>::max();
200  double me = std::numeric_limits<double>::max();
201  OverlapEntry* best = nullptr;
202  uint32_t cLiveEntries = 0;
203 
204  // find combined region and enlargement of every entry and store it.
205  for (uint32_t cChild = 0; cChild < m_children; ++cChild)
206  {
207  if (m_ptrMBR[cChild]->m_endTime <= r.m_startTime) continue;
208 
209  try
210  {
211  entries[cLiveEntries] = new OverlapEntry();
212  }
213  catch (...)
214  {
215  for (uint32_t i = 0; i < cLiveEntries; ++i) delete entries[i];
216  delete[] entries;
217  throw;
218  }
219 
220  entries[cLiveEntries]->m_index = cChild;
221  entries[cLiveEntries]->m_original = m_ptrMBR[cChild];
222  entries[cLiveEntries]->m_combined = m_pTree->m_regionPool.acquire();
223  m_ptrMBR[cChild]->getCombinedRegion(*(entries[cLiveEntries]->m_combined), r);
224  entries[cLiveEntries]->m_oa = entries[cLiveEntries]->m_original->getArea();
225  entries[cLiveEntries]->m_ca = entries[cLiveEntries]->m_combined->getArea();
226  entries[cLiveEntries]->m_enlargement = entries[cLiveEntries]->m_ca - entries[cLiveEntries]->m_oa;
227 
228  if (entries[cLiveEntries]->m_enlargement < me)
229  {
230  me = entries[cLiveEntries]->m_enlargement;
231  best = entries[cLiveEntries];
232  }
233  else if (entries[cLiveEntries]->m_enlargement == me && entries[cLiveEntries]->m_oa < best->m_oa)
234  {
235  best = entries[cLiveEntries];
236  }
237  ++cLiveEntries;
238  }
239 
240 #ifndef NDEBUG
241  if (cLiveEntries == 0)
242  {
243  std::ostringstream s;
244  s << "findLeastOverlap: All entries of node " << m_identifier << " are dead.";
245  throw Tools::IllegalStateException(s.str());
246  }
247 #endif
248 
249  if (me < -std::numeric_limits<double>::epsilon() || me > std::numeric_limits<double>::epsilon())
250  {
251  uint32_t cIterations;
252 
253  if (cLiveEntries > m_pTree->m_nearMinimumOverlapFactor)
254  {
255  // sort entries in increasing order of enlargement.
256  ::qsort(entries, cLiveEntries,
257  sizeof(OverlapEntry*),
258  OverlapEntry::compareEntries);
259  assert(entries[0]->m_enlargement <= entries[m_children - 1]->m_enlargement);
260 
261  cIterations = m_pTree->m_nearMinimumOverlapFactor;
262  }
263  else
264  {
265  cIterations = cLiveEntries;
266  }
267 
268  // calculate overlap of most important original entries (near minimum overlap cost).
269  for (uint32_t cIndex = 0; cIndex < cIterations; ++cIndex)
270  {
271  double dif = 0.0;
272  OverlapEntry* e = entries[cIndex];
273 
274  for (uint32_t cChild = 0; cChild < cLiveEntries; ++cChild)
275  {
276  if (cIndex != cChild)
277  {
278  double f = e->m_combined->getIntersectingArea(*(entries[cChild]->m_original));
279  if (f != 0.0) dif += f - e->m_original->getIntersectingArea(*(entries[cChild]->m_original));
280  }
281  } // for (cChild)
282 
283  if (dif < leastOverlap)
284  {
285  leastOverlap = dif;
286  best = e;
287  }
288  else if (dif == leastOverlap)
289  {
290  if (e->m_enlargement == best->m_enlargement)
291  {
292  // keep the one with least area.
293  if (e->m_original->getArea() < best->m_original->getArea()) best = e;
294  }
295  else
296  {
297  // keep the one with least enlargement.
298  if (e->m_enlargement < best->m_enlargement) best = e;
299  }
300  }
301  } // for (cIndex)
302  }
303 
304  uint32_t ret = best->m_index;
305 
306  for (uint32_t cChild = 0; cChild < cLiveEntries; ++cChild)
307  {
308  delete entries[cChild];
309  }
310  delete[] entries;
311 
312  return ret;
313 }

315 void Index::adjustTree(Node* n, std::stack<id_type>& pathBuffer)
316 {
317  ++(m_pTree->m_stats.m_u64Adjustments);
318 
319  // find entry pointing to old node;
320  uint32_t child;
321  for (child = 0; child < m_children; ++child)
322  {
323  if (m_pIdentifier[child] == n->m_identifier) break;
324  }
325 
326  // MBR needs recalculation if either:
327  // 1. the NEW child MBR is not contained.
328  // 2. the OLD child MBR is touching.
329  bool bContained = m_nodeMBR.containsRegion(n->m_nodeMBR);
330  bool bTouches = m_nodeMBR.touchesRegion(*(m_ptrMBR[child]));
331  bool bRecompute = (! bContained || (bTouches && m_pTree->m_bTightMBRs));
332 
333  // we should not adjust time here
334  double st = m_ptrMBR[child]->m_startTime;
335  double en = m_ptrMBR[child]->m_endTime;
336  *(m_ptrMBR[child]) = n->m_nodeMBR;
337  m_ptrMBR[child]->m_startTime = st;
338  m_ptrMBR[child]->m_endTime = en;
339 
340  if (bRecompute)
341  {
342  // no need to update times here. The inserted MBR is younger than all nodes.
343 
344  for (uint32_t cDim = 0; cDim < m_nodeMBR.m_dimension; ++cDim)
345  {
346  m_nodeMBR.m_pLow[cDim] = std::numeric_limits<double>::max();
347  m_nodeMBR.m_pHigh[cDim] = -std::numeric_limits<double>::max();
348 
349  for (uint32_t cChild = 0; cChild < m_children; ++cChild)
350  {
351  m_nodeMBR.m_pLow[cDim] = std::min(m_nodeMBR.m_pLow[cDim], m_ptrMBR[cChild]->m_pLow[cDim]);
352  m_nodeMBR.m_pHigh[cDim] = std::max(m_nodeMBR.m_pHigh[cDim], m_ptrMBR[cChild]->m_pHigh[cDim]);
353  }
354  }
355  }
356 
357  m_pTree->writeNode(this);
358 
359  if (bRecompute && (! pathBuffer.empty()))
360  {
361  id_type cParent = pathBuffer.top(); pathBuffer.pop();
362  NodePtr ptrN = m_pTree->readNode(cParent);
363  Index* p = static_cast<Index*>(ptrN.get());
364  p->adjustTree(this, pathBuffer);
365  }
366 }

368 void Index::adjustTree(Node* n, Node* nn, std::stack<id_type>& pathBuffer)
369 {
370  ++(m_pTree->m_stats.m_u64Adjustments);
371 
372  // find entry pointing to old node;
373  uint32_t child, child2 = m_capacity;
374  for (child = 0; child < m_children; ++child)
375  {
376  if (m_pIdentifier[child] == nn->m_identifier) child2 = child;
377  if (m_pIdentifier[child] == n->m_identifier) break;
378  }
379 
380  if (child2 == m_capacity)
381  {
382  for (child2 = child + 1; child2 < m_children; ++child2)
383  {
384  if (m_pIdentifier[child2] == nn->m_identifier) break;
385  }
386  }
387 
388  // MBR needs recalculation if either:
389  // 1. the NEW child MBR is not contained.
390  // 2. the OLD child MBR is touching.
391  // 3. the SIBLING MBR is touching.
392  bool b1 = m_nodeMBR.containsRegion(n->m_nodeMBR);
393  bool b2 = m_nodeMBR.touchesRegion(*(m_ptrMBR[child]));
394  bool b3 = m_nodeMBR.touchesRegion(*(m_ptrMBR[child2]));
395  bool bRecompute = (! b1) || ((b2 || b3) && m_pTree->m_bTightMBRs);
396 
397  // we should not adjust time here
398  double st = m_ptrMBR[child]->m_startTime;
399  double en = m_ptrMBR[child]->m_endTime;
400  *(m_ptrMBR[child]) = n->m_nodeMBR;
401  m_ptrMBR[child]->m_startTime = st;
402  m_ptrMBR[child]->m_endTime = en;
403 
404  st = m_ptrMBR[child2]->m_startTime;
405  en = m_ptrMBR[child2]->m_endTime;
406  *(m_ptrMBR[child2]) = nn->m_nodeMBR;
407  m_ptrMBR[child2]->m_startTime = st;
408  m_ptrMBR[child2]->m_endTime = en;
409 
410  if (bRecompute)

413 
414  for (uint32_t cDim = 0; cDim < m_nodeMBR.m_dimension; ++cDim)
415  {
416  m_nodeMBR.m_pLow[cDim] = std::numeric_limits<double>::max();
417  m_nodeMBR.m_pHigh[cDim] = -std::numeric_limits<double>::max();
418 
419  for (uint32_t cChild = 0; cChild < m_children; ++cChild)
420  {
421  m_nodeMBR.m_pLow[cDim] = std::min(m_nodeMBR.m_pLow[cDim], m_ptrMBR[cChild]->m_pLow[cDim]);
422  m_nodeMBR.m_pHigh[cDim] = std::max(m_nodeMBR.m_pHigh[cDim], m_ptrMBR[cChild]->m_pHigh[cDim]);
